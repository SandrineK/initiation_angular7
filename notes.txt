ng generate component animal\detail-animal --module=Animal

#Etape 1 : récupérer l'id 
detail compotent ts : ajouter dans constructeur ==> private route: ActivatedRoute

# récupérer l'id de la route et l'afficher dans la console
const id = +this.route.snapshot.paramMap.get('id');

# Effectuer l'injection de dépendance du service animal
private animalService : AnimalService

#stocker en local l'animal
this.animal = this.animalService.get(id);

#html de détail :
Afficher le nom ==> <h2>Fiche de {{animal.nom | uppercase}}</h2>
